MvptL(pL):=(
  apply(pL,Mvpt(#));
);

pA=[0,0];
pB=[sqrt(3),1];
pC=[sqrt(3),0];
Listplot("1",MvptL([pA,pB,pC,pA]));
Expr(Mvpt((pA+pB)/2),"nw","2");
Expr(Mvpt((pB+pC)/2),"e","1");
Expr(Mvpt((pC+pA)/2),"s","\sqrt{3}");
Letter(Mvpt(pA),"w","A");
Letter(Mvpt(pB),"n","B");
Letter(Mvpt(pC),"e","C");
Paramark("mv2",MvptL([pA,pC,pB]),[4]);
